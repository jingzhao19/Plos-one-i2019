 proc sort; by name ; run;
proc mixed data=want;
      by name;
      class Trt Time subject;

	  lsmeans Trt*time / slice=time; 
	  ods output Tests3 = main;
      ods output Slices = time;
run;
proc export data=main dbms=xlsx
outfile="C:\Jing STATS\creative component\2018\simulation\5 time point simulation\5 point alter model\arh\result_arh_arh_r1n3_main.csv" dbms=csv
replace;
run;

  array scores &scores ;
  do i=1 to dim(scores);
    scoret2=scores(i);
    name=vname(scores(i));
    output;
  end;
 run;
 proc sort; by name ; run;
proc glm data=t2;
      by name;
      class Trt;
      model scoret2 = Trt;
	  ods output  overallANOVA = tp2;
run;
proc export data=tp2 dbms=xlsx
outfile="C:\Jing STATS\creative component\2018\simulation\5 time point simulation\5 point alter model\arh\result_arh_anova_r9n12_T2.csv" dbms=csv
replace;
run;
