/******************SAS code for simulated data for null model (Simulation: Heterogenous compound symmetry, r=0.1 and n=3*/

/*RM analysis using heterogenous compound symmetry (csh)*/
data nullcsh_csh_r1n3;
infile 'C:\Jing STATS\creative component\2018\simulation\5 time point simulation\5 point null model\CSH\Null csh model_r0.1_n3.csv' delimiter=',' MISSOVER FIRSTOBS=2;
input Trt Subject Time X1-X1000;
run;
%let scores = X1-X1000;
data want ; set nullcsh_csh_r1n3;
  array scores &scores ;
  do i=1 to dim(scores);
    score=scores(i);
    name=vname(scores(i));
    output;
  end;
 run;
 proc sort; by name ; run;
proc mixed data=want;
      by name;
      class Trt Time subject;
      model score = Trt | time;
      repeated Time/subject=subject(trt) type=csh;
	  lsmeans Trt*time / slice=time; 
	  ods output Tests3 = main;
      ods output Slices = time;
run;
proc export data=main dbms=xlsx
outfile="C:\Jing STATS\creative component\2018\simulation\5 time point simulation\5 point null model\CSH\ddfm=Default\result_nullcsh_csh_r1n3_main.csv" dbms=csv
replace;
run;
proc export data=time dbms=xlsx
outfile="C:\Jing STATS\creative component\2018\simulation\5 time point simulation\5 point null model\CSH\ddfm=Default\result_nullcsh_csh_r1n3_slice.csv" dbms=csv
replace;
run;
